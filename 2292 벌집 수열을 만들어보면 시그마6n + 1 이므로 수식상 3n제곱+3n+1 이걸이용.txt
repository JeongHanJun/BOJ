import java.util.*;
import java.io.*;
public class Main {
    public static void main(String[] args) throws IOException {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int k = 1;
        for( int i = 0 ; i < n ; i++)
        {
            k = 3*i*i + 3*i + 1 ;
            if(k >= n)
            {
                System.out.println(i+1);
                break;
            }
        }
        sc.close();
    }
}
